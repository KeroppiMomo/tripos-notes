$$ \begin{align*} \prob(X\in B|N=n) &= \frac{\prob(V_k > f(U_k)/\lambda \text{ for } k =1,\ldots,n-1 \text{ and } V_n \leq f(U_n)/\lambda \text{ and } U_n \in B)}{\prob(V_k > f(U_k)/\lambda \text{ for } k =1,\ldots,n-1 \text{ and } V_n \leq f(U_n)/\lambda)} \\ &= \frac{\prob(V_n \leq f(U_n)/\lambda \text{ and } U_n \in B)}{\prob(V_n\leq f(U_n)/\lambda)} &&\text{by independence} \\ &= \frac{\int_{[0,1]^d} 1_{\set{u\in B}} \int_{0}^{1} 1_{\set{v\leq f(u)/\lambda}} \, dv \, du}{\int_{[0,1]^d} \int_{0}^{1} 1_{\set{v\leq f(u)/\lambda}} \, dv \, du} \\ &= \frac{\int_{B} \frac{f(u)}{\lambda} du}{\int_{[0,1]^d} \frac{f(u)}{\lambda} du} \\ &= \int_{B} f(u)\, du \end{align*} $$ so by the law of total probability, $$ \begin{align*} \prob(X\in B) &= \sum_{n=1}^{\infty} \int_{B} f(u)\, du \prob(N=n) = \int_{B} f(u)\,du. \end{align*} $$
