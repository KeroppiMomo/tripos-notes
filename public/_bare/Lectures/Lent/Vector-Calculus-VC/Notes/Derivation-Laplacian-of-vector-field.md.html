$$\begin{align*} \big(\nabla(\nabla\cdot\v F) - \nabla\times(\nabla\times\v F)\big)_i &= \pdfd{(\nabla\cdot\v F)}{x^i}\v - \epsilon_{ijk}\pdfd{(\nabla\times\v F)_k}{x^j} \\ &= \frac{\partial^2F_j}{\partial x^i\partial x^j} - \epsilon_{ijk}\epsilon_{pqk}\frac{\partial^2 F_q}{\partial x^j\partial x^p} \\ &= \frac{\partial^2F_j}{\partial x^i\partial x^j} - (\delta_{ip}\delta_{jq} - \delta_{iq}\delta_{jp})\frac{\partial^2 F_q}{\partial x^j\partial x^p} \\ &= \frac{\partial^2F_j}{\partial x^i\partial x^j} - \frac{\partial^2 F_j}{\partial x^j\partial x^i} + \frac{\partial^2 F_i}{\partial x^j\partial x^j} \\ &= \frac{\partial^2 F_i}{\partial x^j\partial x^j} \\ &= (\nabla^2\v F)_i \end{align*}$$
