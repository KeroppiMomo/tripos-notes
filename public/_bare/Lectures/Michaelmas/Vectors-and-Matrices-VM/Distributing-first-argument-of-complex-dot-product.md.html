$$\begin{align*} (\lambda\vec u +\mu\vec v)\cdot\vec w &= \overline{\vec w\cdot(\lambda\vec u+\mu\vec v)} \\ &= \overline{\lambda\vec w\cdot\vec u + \mu\vec w\cdot\vec v} \\ &= \overline{\lambda}\,\overline{\vec w\cdot\vec u} + \overline{\mu}\overline{\vec w\cdot\vec v} \\ &= \overline\lambda \vec u\cdot\vec w + \overline\mu\vec v\cdot\vec w. \end{align*}$$
